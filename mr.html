/** * ระบบบันทึกอุบัติการณ์ ฝ่ายเภสัชกรรม รพ.สวี * รวมการจัดการข้อมูล ER และ MR ไว้ใน Script เดียว (เวอร์ชันปรับปรุงป้องกัน Error) */ // --- CONFIGURATION --- const SS_ID_ER = "1clByvjX7xVl0FjcF3oHw0-AiUX9Ubk2oRu2GgrRXPCU"; const SS_ID_MR = "1c3DXVSB0ZqJJWEE-EL3q-xrS9PMbKBqJgKBpk4NVOpQ"; const SS_ID_DRUG = "1iO7pGLGGoK3zkZE4ckr5HofV-6rBeoVMCWZZOawhEyY"; const SHEETS = { ER_HISTORY: "ER ERROR", // ⚠️ ต้องตรงกับชื่อแท็บในไฟล์ ER เป๊ะๆ MR_HISTORY: "MR ERROR", // ⚠️ ต้องตรงกับชื่อแท็บในไฟล์ MR เป๊ะๆ MR_CATEGORY: "ERROR CATEGORY", DRUG_LIST: "full_drugcode" }; function doGet(e) { try { const ssEr = SpreadsheetApp.openById(SS_ID_ER); const ssMr = SpreadsheetApp.openById(SS_ID_MR); const ssDrug = SpreadsheetApp.openById(SS_ID_DRUG); // 1. ดึงข้อมูลยา (Col C) พร้อมระบบป้องกันถ้าหาชีตไม่เจอ let drugs = []; const drugSheet = ssDrug.getSheetByName(SHEETS.DRUG_LIST); if (drugSheet) { // ดึงจากคอลัมน์ C (3) และกรองค่าว่างออก drugs = drugSheet.getRange(2, 3, drugSheet.getLastRow() - 1, 1).getValues().flat().filter(String); } // 2. ดึงข้อมูล MR Category let mrCategories = []; const catSheet = ssMr.getSheetByName(SHEETS.MR_CATEGORY); if (catSheet && catSheet.getLastRow() > 1) { mrCategories = catSheet.getRange(2, 1, catSheet.getLastRow() - 1, 3).getValues().map(r => ({ id: r[0], type: r[1], error: r[2] })); } // 3. ดึงข้อมูลประวัติ (History) รวมทั้ง 2 ฝั่งเพื่อแสดงผล Dashboard const erHistory = getSheetData(ssEr.getSheetByName(SHEETS.ER_HISTORY), "ER"); const mrHistory = getSheetData(ssMr.getSheetByName(SHEETS.MR_HISTORY), "MR"); // รวมประวัติ const allHistory = [...erHistory, ...mrHistory]; // 4. คำนวณ ERROR ID ล่าสุดของ MR let lastId = 0; const mrHistorySheet = ssMr.getSheetByName(SHEETS.MR_HISTORY); if (mrHistorySheet && mrHistorySheet.getLastRow() > 1) { lastId = Number(mrHistorySheet.getRange(mrHistorySheet.getLastRow(), 1).getValue()) || 0; } // 5. ข้อมูลพื้นฐานสำหรับ Dropdown (ดึงจากประวัติเก่า) const result = { status: "success", drugs: drugs, errorCategories: mrCategories, inspectors: getUniqueColumn(ssEr.getSheetByName(SHEETS.ER_HISTORY), 11), // Col 11 คือผู้ตรวจ doctors: getUniqueColumn(mrHistorySheet, 12), // Col 12 (L) ใน MR recorders: getUniqueColumn(mrHistorySheet, 13), // Col 13 (M) ใน MR lastId: lastId, history: allHistory }; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString(), hint: "โปรดตรวจสอบ 1.ชื่อ Sheet ถูกต้องไหม 2.กดยืนยันสิทธิ์หรือยัง" })).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const p = JSON.parse(e.postData.contents); // โหมดลบข้อมูล if (p.action === "DELETE") { return deleteData(p); } // โหมดบันทึกข้อมูล const mode = p.systemMode; if (mode === "ER") { return saveErData(p); } else { return saveMrData(p); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // --- HELPER FUNCTIONS --- function deleteData(p) { const ss = p.systemMode === "ER" ? SpreadsheetApp.openById(SS_ID_ER) : SpreadsheetApp.openById(SS_ID_MR); const sheetName = p.systemMode === "ER" ? SHEETS.ER_HISTORY : SHEETS.MR_HISTORY; const sheet = ss.getSheetByName(sheetName); if (!sheet) throw new Error("ไม่พบชีตชื่อ " + sheetName); const rows = sheet.getDataRange().getValues(); // ค้นหาแถวที่ตรงกับ ID ที่ส่งมา (คอลัมน์แรก index 0) for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(p.id)) { sheet.deleteRow(i + 1); // +1 เพราะ index ของ getValues เริ่มจาก 0 แต่ deleteRow เริ่มจาก 1 return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "ไม่พบข้อมูลที่ต้องการลบ" })).setMimeType(ContentService.MimeType.JSON); } function saveErData(p) { const ss = SpreadsheetApp.openById(SS_ID_ER); const sheet = ss.getSheetByName(SHEETS.ER_HISTORY); if (!sheet) throw new Error("ไม่พบชีตชื่อ " + SHEETS.ER_HISTORY); // คำนวณค่า Col C จำนวน Error (1 หรือ 0) const errorCount = p.errorType === "❌ไม่พบ Error" ? 0 : 1; // คำนวณลำดับที่ (ถ้าเป็น - ให้ส่งค่า 0 ไปแทน) const seq = p.sequence === '-' ? 0 : p.sequence; // แปลงวันที่สำหรับ Col N เป็น DD/MM/YYYY const d = new Date(p.date); const formattedDate = Utilities.formatDate(d, "GMT+7", "dd/MM/yyyy"); // บันทึกลงเรียงตาม Column A -> N (และต่อด้วย O, P) sheet.appendRow([ p.shiftDate, // Col A: กะ (เวรรับยา) p.totalRx, // Col B: จำนวนใบสั่งยา errorCount, // Col C: จำนวน Error seq, // Col D: ลำดับที่ p.time, // Col E: เวลา p.errorType, // Col F: ความเสี่ยงที่พบ p.details, // Col G: รายละเอียด p.riskType, // Col H: ประเภทความเสี่ยง p.severity, // Col I: ระดับ p.action, // Col J: การแก้ไข p.inspector, // Col K: ผู้ตรวจเช็ค p.doctor, // Col L: รคส. p.recorder, // Col M: ผู้บันทึก formattedDate, // Col N: วันที่ (DD/MM/YYYY) p.hn, // Col O: HN p.drugs, // Col P: รายการยา "ER" // Col Q: System Mode Flag ]); return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON); } function saveMrData(p) { const ss = SpreadsheetApp.openById(SS_ID_MR); const sheet = ss.getSheetByName(SHEETS.MR_HISTORY); if (!sheet) throw new Error("ไม่พบชีตชื่อ " + SHEETS.MR_HISTORY); // คำนวณ ERROR ID ล่าสุด const lastRow = sheet.getLastRow(); const nextId = lastRow > 1 ? Number(sheet.getRange(lastRow, 1).getValue()) + 1 : 1; // แปลงวันที่ const d = new Date(p.date); const formattedDate = Utilities.formatDate(d, "GMT+7", "dd/MM/yyyy"); // สำหรับ Col B const monthStr = Utilities.formatDate(d, "GMT+7", "yyyy-MM"); // สำหรับ Col R sheet.appendRow([ nextId, // Col A: ลำดับที่ (ERROR ID) formattedDate, // Col B: วันที่ (DD/MM/YYYY) p.type, // Col C: ความคลาดเคลื่อน (Category) p.error, // Col D: รายการ Error p.details, // Col E: รายละเอียด (Auto Text จากหน้าเว็บ) p.department, // Col F: หน่วยงานที่เกิด p.severity, // Col G: ระดับความรุนแรง p.policy, // Col H: นโยบายยา p.hn, // Col I: HN p.drug1, // Col J: ยา 1 p.drug2, // Col K: ยา 2 p.doctor, // Col L: แพทย์ p.recorder, // Col M: เภสัชกร p.qty, // Col N: จำนวน (เม็ด) "", // Col O: ชื่อผู้ป่วย "", // Col P: โทรศัพท์ "Active", // Col Q: สถานะ monthStr, // Col R: รายเดือน "MR" // Col S: System Mode Flag (ใช้แยกข้อมูลตอนดึงกลับ) ]); return ContentService.createTextOutput(JSON.stringify({ status: "success", id: nextId })).setMimeType(ContentService.MimeType.JSON); } function getSheetData(sheet, mode) { if (!sheet) return []; const lastRow = sheet.getLastRow(); if (lastRow <= 1) return []; const rows = sheet.getDataRange().getValues(); const headers = rows[0]; // ดึง 300 รายการล่าสุดเพื่อความรวดเร็วในการแสดงผล const startRow = Math.max(1, rows.length - 300); const data = []; for (let i = rows.length - 1; i >= startRow; i--) { const obj = {}; headers.forEach((h, idx) => { let val = rows[i][idx]; // ป้องกันเวลาเพี้ยนเป็น 1899 if (val instanceof Date) { if (val.getFullYear() === 1899) { val = Utilities.formatDate(val, "GMT+7", "HH:mm"); } else { val = Utilities.formatDate(val, "GMT+7", "yyyy-MM-dd"); } } // ป้องกัน Error จากคอลัมน์ที่เว้นว่างไว้ หรือมีการเคาะ Spacebar ผิดใน Header if (h) { obj[h.toString().trim()] = val; } }); obj.systemMode = mode; // บังคับใส่ Mode เพื่อให้ UI หน้าเว็บแยกสถิติถูก data.push(obj); } return data; } function getUniqueColumn(sheet, colIndex) { if (!sheet || sheet.getLastRow() < 2) return []; const vals = sheet.getRange(2, colIndex, sheet.getLastRow() - 1, 1).getValues().flat(); return [...new Set(vals)].filter(String).map(s => s.toString().trim()).sort(); }
